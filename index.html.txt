    <script>
        function switchTab(tabName) {
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            document.querySelectorAll('.content-section').forEach(sec => sec.classList.remove('active'));

            if (tabName === 'info') {
                document.querySelectorAll('.tab-btn')[0].classList.add('active');
                document.getElementById('section-info').classList.add('active');
            } else if (tabName === 'segnala') {
                document.querySelectorAll('.tab-btn')[1].classList.add('active');
                document.getElementById('section-segnala').classList.add('active');
            }
        }

        function inviaSegnalazioneViaMail(event) {
            event.preventDefault();

            const data = document.getElementById('dataInput').value;
            const plesso = document.getElementById('plessoSelect').value;
            const ruolo = document.getElementById('ruoloSelect').value;
            const descrizione = document.getElementById('descrizioneText').value;

            let tipiSelezionati = [];
            document.querySelectorAll('.tipoEpisodio:checked').forEach(cb => {
                tipiSelezionati.push(cb.value);
            });
            const comportamenti = tipiSelezionati.length > 0 ? tipiSelezionati.join(', ') : 'Non specificato';

            const emailDestinatario = "team.antibullismo@magnaghisolari.edu.it";
            const oggetto = encodeURIComponent("Modulo 1 - Nuova Segnalazione Presunto Caso");
            
            let corpoMail = `MODULO 1 - SEGNALAZIONE CASO DI BULLISMO / CYBERBULLISMO\n`;
            corpoMail += `==================================================\n\n`;
            corpoMail += `Data dell'episodio: ${data}\n`;
            corpoMail += `Plesso Scolastico: ${plesso}\n`;
            corpoMail += `Segnalazione inviata da: ${ruolo}\n`;
            corpoMail += `Tipologia di comportamenti: ${comportamenti}\n\n`;
            corpoMail += `DESCRIZIONE DELL'ACCADUTO:\n`;
            corpoMail += `${descrizione}\n\n`;
            corpoMail += `==================================================\n`;
            corpoMail += `Inviato tramite Applicazione Team Antibullismo.`;

            const corpoMailFormattato = encodeURIComponent(corpoMail);
            window.location.href = `mailto:${emailDestinatario}?subject=${oggetto}&body=${corpoMailFormattato}`;
        }
    </script>
